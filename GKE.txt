0- Mettre à jour gcloud et installer le plugin gcloud pour l'authentification

gclout components update
gcloud components install gke-gcloud-auth-plugin
echo 'export USE_GKE_GCLOUD_AUTH_PLUGIN=True' >> ~/.bashrc
source ~/.bashrc


1- Create project

gcloud projects create [PROJECT_NAME]
gcloud config set project [PROJECT_NAME]

2- Activate API

gcloud services enable container.googleapis.com

3- Create a cluster in Autopilot mode

gcloud container clusters create-auto [CLUSTER_NAME] --region=europe-west1

4- Récupérer les informations d'identification

gcloud container clusters get-credentials [CLUSTER_NAME] --region europe-west1

5 - Gérer les règles de firewall

Exemple de règle:

gcloud compute firewall-rules create allow-http \
  --direction=INGRESS \
  --priority=1000 \
  --network=default \
  --action=ALLOW \
  --rules=tcp:80 \
  --source-ranges=0.0.0.0/0 \
  --target-tags=http-server

-------------------------------------------

1- Remove cluster

gcloud container clusters delete [CLUSTER_NAME] --region=europe-west1

2- Remove volumes

gcloud compute disks list
gcloud compute disks delete [DISK_NAME] --zone=[ZONE_NAME]

3- Remove IPs

gcloud compute addresses list
gcloud compute addresses delete [IP_NAME] --region=[REGION]

4- Remove firewall rules

gcloud compute firewall-rules list
gcloud compute firewall-rules delete [RULE_NAME]

OR

1 - Simply remove project

gcloud projects delete [PROJECT_ID]
